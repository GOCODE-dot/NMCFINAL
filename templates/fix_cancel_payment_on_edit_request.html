            cash_check = db.execute(
                """SELECT id FROM cash_payment_requests
                   WHERE student_id = ?
                     AND status     = 'cancelled'
                     AND ABS(strftime('%s', updated_at) -
                             strftime('%s', ?)) < 30
                   LIMIT 1""",
                (r['student_id'], r['created_at'])
            ).fetchone()
            # Check bKash
            bkash_check = db.execute(
                """SELECT id FROM payments
                   WHERE student_id      = ?
                     AND payment_status  = 'cancelled'
                     AND ABS(strftime('%s', updated_at) -
                             strftime('%s', ?)) < 30
                   LIMIT 1""",
                (r['student_id'], r['created_at'])
            ).fetchone()
            r['payment_cancelled'] = bool(cash_check or bkash_check)
            r['payment_type']      = 'bKash' if bkash_check else ('Cash' if cash_check else None)
        else:
            r['payment_cancelled'] = False
            r['payment_type']      = None
        enriched.append(r)

    return jsonify(requests=enriched)
